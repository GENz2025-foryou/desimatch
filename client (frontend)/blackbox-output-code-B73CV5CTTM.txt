import { useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { motion } from 'framer-motion';
import { PhoneIcon, CameraIcon } from '@heroicons/react/24/outline';
import toast from 'react-hot-toast';
import api from '../services/api';

const Signup = () => {
  const [step, setStep] = useState(1);
  const [phone, setPhone] = useState('');
  const [otp, setOtp] = useState('');
  const [loading, setLoading] = useState(false);
  const navigate = useNavigate();

  const sendOTP = async () => {
    if (!phone.startsWith('+91') || phone.length !== 13) {
      toast.error('Please enter valid Indian number (+91)');
      return;
    }

    setLoading(true);
    try {
      await api.post('/auth/send-otp', { phone });
      toast.success('OTP sent to your phone!');
      setStep(2);
    } catch (error) {
      toast.error(error.response?.data?.error || 'Error sending OTP');
    } finally {
      setLoading(false);
    }
  };

  const verifyOTP = async () => {
    setLoading(true);
    try {
      const response = await api.post('/auth/verify-otp', { phone, otp });
      localStorage.setItem('token', response.data.token);
      localStorage.setItem('userId', response.data.userId);
      toast.success('Phone verified! Now verify your face.');
      navigate('/verify-face');
    } catch (error) {
      toast.error(error.response?.data?.error || 'Invalid OTP');
    } finally {
      setLoading(false);
    }
  };

    <div className="min-h-screen flex items-center justify-center p-6 bg-gradient-to-br from-slate-50 to-blue-50 dark:from-gray-900 dark:to-gray-800">
      <motion.div

        className="glass-card w-full max-w-md p-10"
